# install docker dan docker compose
```
sudo pacman -S docker docker-compose
```

# cloning repository
masuk ke github slims
```
git clone https://github.com/slims/docker-compose-for-slims.git
```

# masuk ke folder docker compose
```
cd docker-compose-for-slims/
```


# running docker
```
sudo docker-compose up -d
```

# ganti kepemilikan
```
chmod -R 777 app/
```

# running slims
```
localhost/slims/index.php
```
